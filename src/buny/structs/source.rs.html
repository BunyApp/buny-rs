117/// A result of a deep link handling.
118#[derive(Debug, Clone, PartialEq)]
119pub enum DeepLinkResult {
120	Novel { key: String },
121	Chapter { novel_key: String, key: String },
122	Listing(Listing),
123}
124
125impl Serialize for DeepLinkResult {
126	fn serialize<S>(&self, serializer: S) -> core::result::Result<S::Ok, S::Error>
127	where
128		S: serde::Serializer,
129	{
130		let mut state = serializer.serialize_struct("DeepLinkResult", 3)?;
131		match self {
132			DeepLinkResult::Novel { key } => {
133				state.serialize_field("novel_key", &Some(key))?;
134				state.serialize_field("chapter_key", &Option::<String>::None)?;
135				state.serialize_field("listing", &Option::<Listing>::None)?;
136			}
137			DeepLinkResult::Chapter { novel_key, key } => {
138				state.serialize_field("novel_key", &Some(novel_key))?;
139				state.serialize_field("chapter_key", &Some(key))?;
140				state.serialize_field("listing", &Option::<Listing>::None)?;
141			}
142			DeepLinkResult::Listing(listing) => {
143				state.serialize_field("novel_key", &Option::<String>::None)?;
144				state.serialize_field("chapter_key", &Option::<String>::None)?;
145				state.serialize_field("listing", &Some(listing))?;
146			}
147		}
148		state.end()
149	}
150}
